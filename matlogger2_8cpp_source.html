189 int MatLogger2::flush_available_data()
190 {
191  // number of flushed bytes is returned on exit
192  int bytes = 0;
193 
194  // acquire exclusive access to the _vars object
195  std::lock_guard<MutexType> lock(_vars_mutex->get());
196  for(auto& p : _vars)
197  {
198  Eigen::MatrixXd block;
199  int valid_elems = 0;
200 
201  // while there are blocks available for reading..
202  while(p.second.read_block(block, valid_elems))
203  {
204  // get rows & cols for a single sample
205  auto dims = p.second.get_dimension();
206 
207  int rows = -1;
208  int cols = -1;
209  int slices = -1;
210  bool is_vector = false;
211 
212 
213  if(dims.second == 1) // the variable is a vector
214  {
215  rows = dims.first;
216  cols = valid_elems;
217  slices = 1;
218  is_vector = true;
219  }
220  else // the variable is a matrix
221  {
222  rows = dims.first;
223  cols = dims.second;
224  slices = valid_elems;
225  is_vector = false;
226  }
227 
228  // write block
229  _backend->write(p.second.get_name().c_str(),
230  block.data(),
231  rows, cols, slices);
232 
233  // update bytes computation
234  bytes += block.rows() * valid_elems * sizeof(double);
235  }
236  }
237 
238  return bytes;
239 }
